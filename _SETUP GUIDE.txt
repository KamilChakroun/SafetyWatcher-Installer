# Frontend
VITE_WS_URL=ws://localhost/ws

STREAMER_URLS=ws://streamer:9000

# Auth
SECRET_KEY= PASTE-SECRET-KEY-HERE
ALGORITHM=HS256
ACCESS_TOKEN_EXPIRE_MINUTES=480

# CORS
ALLOWED_ORIGINS=http://localhost

# Metrics (Leave empty for now)
METRICS_TOKEN=

# Camera database
MONGO_CAM_DB=surveillance
MONGO_CAM_COLL=cameras

# Minio DB (change access key & secret key to whatever you like)
MINIO_ACCESS_KEY=minioadmin
MINIO_SECRET_KEY=minioadmin
MINIO_BUCKET=safety-watcher-history
MINIO_PRESIGN_ENDPOINT=localhost:9000

EVENT_WINDOW_SECONDS=120
